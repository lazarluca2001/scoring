async function upload() { const btn = document.getElementById('up-btn'); const event = document.getElementById('event-name').value; const cat = document.getElementById('cat-name').value; const text = document.getElementById('raw-input').value; if(!event || !cat || !text) return alert("Minden mezőt tölts ki!"); // Szeletelés a # alapján const parts = text.split('#').filter(p => p.trim() !== ""); // Táncosok kinyerése let dancers = parts.map(p => { const bibMatch = p.match(/^(\d+)/); if(!bibMatch) return null; const bib = bibMatch[1]; // Eltávolítjuk a Bib-et és a végén ragadt "Pos" számot is let rawName = p.replace(bib, '').trim(); let cleanName = rawName.replace(/\d+$/, '').trim(); // Szerepkör eldöntése a pozíció vagy szöveg alapján // Ha az Admin-ba beillesztett szövegben benne van a "Follower" szó let role = p.toLowerCase().includes('follower') ? 'Follower' : 'Leader'; return { bib: bib, name: cleanName, role: role }; }).filter(d => d !== null); btn.disabled = true; btn.innerText = "Küldés..."; try { await fetch(SCRIPT_URL, { method: "POST", mode: "no-cors", // A biztonság kedvéért marad, ha a táblázat frissül, akkor jó body: JSON.stringify({ action: "upload_dancers", eventName: event, category: cat, data: dancers }) }); alert("Lista elküldve a(z) " + event + " lapra!"); } catch (e) { alert("Hiba történt!"); console.error(e); } finally { btn.disabled = false; btn.innerText = "ADATOK HOZZÁADÁSA"; } }
